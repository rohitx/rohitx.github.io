# Create CSV file for analysis in Pandas. Write select parameters to a file: 
file = open('SDSS_survey.csv',"w")
for i in range(len(stars['TARGET_ID'])):
    this_apogee_id = stars['APOGEE_ID'][i]
    this_locid    = stars['LOCATION_ID'][i]
    this_field     = stars['FIELD'][i]
    this_jmag      = stars['J'][i]
    this_hmag      = stars['H'][i]
    this_kmag      = stars['K'][i]
    this_ra        = stars['RA'][i]
    this_de        = stars['DEC'][i]
    this_teff      = stars['RV_TEFF'][i]
    this_logg      = stars['RV_LOGG'][i]
    this_snr       = stars['SNR'][i]
    this_pmra      = stars['PMRA'][i]
    this_pmde      = stars['PMDEC'][i]
    file.write("{this_apogee_id:s}, {this_locid:.2f}, {this_pmra:.6f}, {this_pmde:.6f}, \
               {this_field:s}, {this_jmag:.3f}, {this_hmag:.3f}, {this_kmag:.3f}, {this_ra:.6f},\
               {this_de:.6f}, {this_teff:.2f}, {this_logg:.3f}, {this_snr:.2f}\n \
               ".format(this_apogee_id = this_apogee_id, this_locid=this_locid, \
                        this_pmra = this_pmra, this_pmde = this_pmde, \
                        this_field = this_field, this_jmag = this_jmag, this_hmag = this_hmag, \
                        this_kmag = this_kmag, this_ra = this_ra, this_de = this_de, \
                        this_teff = this_teff, this_logg = this_logg, this_snr = this_snr))
file.close()
